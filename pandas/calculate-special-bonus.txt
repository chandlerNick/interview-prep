# 1873. Calculate Special Bonus

import pandas as pd

def calculate_special_bonus(employees: pd.DataFrame) -> pd.DataFrame:
    employees["bonus"] = employees.salary * (~(employees.name.str.startswith("M")) & (employees.employee_id % 2 != 0))
    return employees.drop(columns=["name", "salary"]).sort_values("employee_id")
