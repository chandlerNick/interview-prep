# 1667. Fix Names in a Table

import pandas as pd

def fix_names(users: pd.DataFrame) -> pd.DataFrame:
    users.name = users.name.apply(lambda x: x[0].upper() + x[1:].lower())
    return users.sort_values(by="user_id")
