# 2890. Reshape Data: Melt

import pandas as pd

def meltTable(report: pd.DataFrame) -> pd.DataFrame:
    return pd.melt(report, id_vars=["product"] , value_vars= ["quarter_1", "quarter_2", "quarter_3", "quarter_4"], var_name="quarter", value_name="sales")


# Notes

# melt function -> inverse of pivot (wide to long)

# id_vars -> cols unchanged
# value_vars -> cols to become rows
# var_name -> var column name
# value_name -> column storing values
