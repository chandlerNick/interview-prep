# 2877. Create a DataFrame from List

import pandas as pd

def createDataframe(student_data: List[List[int]]) -> pd.DataFrame:
    return pd.DataFrame(student_data, columns=["student_id", "age"])

    # Time complexity:
    #
    # n -> num records
    #
    # O(n)

    # Space complexity:
    #
    # c -> num cols
    # n -> num rows
    #
    # O(n*c)
