# 283. Move Zeros


# Naive soln
# O(n^2) -> del shifts "array" elements
# O(k) space k = num zeros

class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        indices = []  # No addiitonal memory?
        for i in range(len(nums)):
            if nums[i] == 0:
                indices.append(i)
        
        zeros = 0
        total_zeros = len(indices)
        for i in indices:
            del nums[i - zeros]
            nums.append(0)
            zeros += 1

# Optimal soln:
class Solution:
	def moveZeros(self, nums: List[int]) -> None:
		last_non_zero = 0
		
		for i in range(len(nums)):
			if nums[i] != 0:
				nums[last_non_zero], nums[i] = nums[i], nums[last_non_zero]
				last_non_zero += 1
