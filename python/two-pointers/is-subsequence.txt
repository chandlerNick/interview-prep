# 392. Is Subsequence

# My soln:

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:

        match_count = 0
        for i in range(len(t)):
            if match_count == len(s):  # Have we yet exhausted the chars in s?
                return True
            if s[match_count] == t[i]:  # Match
                match_count += 1
        
        # Final Check
        if match_count == len(s):
            return True
        else:
            return False

# Runtime: O(n) -- you loop the length of t once.
# Addl Space: O(1) -- not saving anything aside from one int.



# Terser style:

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        if len(s) == 0: return True  # Edge case!
        match_count = 0
        for i in range(len(t)):
            if s[match_count] == t[i]:  # Match
                match_count += 1
                if match_count == len(s):  # Have we yet exhausted the chars in s?
                    return True
        return False
